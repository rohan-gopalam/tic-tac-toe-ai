def checkWin(pos):
    win = False
    winner = ""

    counter = 0
    for i in range(3):
        for j in range(3):
            if pos[i][j] == "":
                counter += 1
    if counter == 0:
        win = True
        winner = "TIE"

    sides = ["X", "O"]
    for side in sides:
        for i in range(3):
            if (pos[i][0] == side and pos[i][1] == side and pos[i][2] == side) or (
                    pos[0][i] == side and pos[1][i] == side and pos[2][i] == side):
                win = True
                winner = side

        if (pos[0][0] == side and pos[1][1] == side and pos[2][2] == side) or (
                pos[0][2] == side and pos[1][1] == side and pos[2][0] == side):
            win = True
            winner = side

    return win, winner
